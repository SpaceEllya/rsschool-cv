function createPhoneNumber(numbers) {
    let firstPart = ""
    let secondPart = ""
    let thirdPart = ""
    for (let  i = 0; i < numbers.length; i++) {
        if (i < 3) {
            firstPart += numbers[i].toString()
        } else if (i >= 3 && i < 6) {
            secondPart += numbers[i].toString()
        } else if (i >= 6) {
            thirdPart += numbers[i].toString()
        }
    }

}
